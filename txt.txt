That's a good next step for PixLite.

But before implementing spatial hashing, make sure you actually need it.

For a small game:

10 sprites
20 enemies
5 bullets

a simple collision loop is fine:

for (const a of colliders) {
    for (const b of colliders) {
        // collision check
    }
}

Spatial hashing becomes useful when you have:

100+ objects
hundreds of bullets
particles
large maps


---

Step 1: Add a Hitbox component

I'd start with a simple AABB (Axis-Aligned Bounding Box).

class Hitbox {
    constructor(
        public parent: Sprite,
        public offsetX = 0,
        public offsetY = 0,
        public width = 16,
        public height = 16
    ) {}

    get x() {
        return this.parent.x + this.offsetX;
    }

    get y() {
        return this.parent.y + this.offsetY;
    }
}

Usage:

player.hitbox = new Hitbox(
    player,
    4,
    2,
    24,
    30
);

Step 2: Collision test

function intersects(
    a: Hitbox,
    b: Hitbox
) {
    return (
        a.x < b.x + b.width &&
        a.x + a.width > b.x &&
        a.y < b.y + b.height &&
        a.y + a.height > b.y
    );
}

Step 4: Hash structure

class SpatialHash {
    cells = new Map<
        string,
        Hitbox[]
    >();
}


---

Insert object

insert(hitbox: Hitbox) {
    const cellX =
        Math.floor(hitbox.x / 64);

    const cellY =
        Math.floor(hitbox.y / 64);

    const key =
        `${cellX},${cellY}`;

    if (!this.cells.has(key)) {
        this.cells.set(key, []);
    }

    this.cells.get(key)!.push(hitbox);
}


---

Query nearby objects

query(hitbox: Hitbox) {
    const cellX =
        Math.floor(hitbox.x / 64);

    const cellY =
        Math.floor(hitbox.y / 64);

    const key =
        `${cellX},${cellY}`;

    return this.cells.get(key) ?? [];
}

Now:

const nearby =
    spatialHash.query(player.hitbox);

for (const other of nearby) {
    if (
        other !== player.hitbox &&
        intersects(
            player.hitbox,
            other
        )
    ) {
        // collision
    }
}


---

Important improvement

A hitbox may span multiple cells.

Example:

+----+----+
| A  | A  |
+----+----+

If you only insert into one cell, collisions can be missed.

Eventually you'll want:

minCellX
maxCellX

minCellY
maxCellY

and insert into every covered cell.

How I'd structure PixLite

Scene
 ├─ SpatialHash
 ├─ Player
 ├─ Enemy
 └─ Bullet

Sprite
 └─ Hitbox

Every frame:

clear spatial hash
↓
insert hitboxes
↓
query nearby
↓
collision checks

A nice API could look like:

player.hitbox = {
    x: 4,
    y: 2,
    width: 24,
    height: 30
};

and internally PixLite creates/manages the actual hitbox and spatial hash entries. This keeps the user API simple while letting you optimize later.

import { Scene } from "./scene"

export class Sprite {
    x: number
    y: number
    atlas: atlas
    imgSrc: string
    scale: number
    flip: { x: boolean, y: boolean }

    private animArray: { key: string, imgSrc: string, atlasArray: atlas[] }[]
    private prevAnimKey: string
    private scene: Scene
    constructor(
        scene: Scene,
        imgSrc: string,
        x: number,
        y: number,
    ) {
        this.scene = scene
        scene.img.src = imgSrc
        this.animArray = []
        this.prevAnimKey = ""

        this.x = x
        this.y = y
        this.scale = 1
        this.atlas = {
            x: 0, y: 0,
            w: scene.img.width,
            h: scene.img.height
        }
        this.imgSrc = imgSrc
        this.flip = { x: false, y: false }
    }

    get width() {
        return this.atlas.w * this.scale
    }

    get height() {
        return this.atlas.h * this.scale
    }

    ctxConfigStatic() {
        this.scene.ctx!.fillStyle = "hsla(0, 100%, 50%, 0.3)"
        const fX = this.flip.x
            ? { c: -1, s: this.width }
            : { c: 1, s: 0 }
        const fY = this.flip.y
            ? { c: -1, s: this.height }
            : { c: 1, s: 0 }
        this.scene.ctx?.translate(fX.s, fY.s)
        this.scene.ctx?.scale(fX.c, fY.c)
    }

    ctxConfigDynamic() {
    }

    redraw(
        callback: (now: number) => void = () => { }
    ) {
        this.scene.img.src = this.imgSrc
        let prevXY: { x: number, y: number }

        this.scene.ctx?.clearRect(
            this.x, this.y,
            this.scene.canvas.width,
            this.scene.canvas.height
        )

        this.ctxConfigStatic()

        this.scene.img.onload = () => {
            this.scene.update((now) => {
                if (prevXY === undefined) prevXY = { x: this.x, y: this.y }

                this.scene.ctx?.clearRect(
                    prevXY.x, prevXY.y,
                    this.width, this.height
                )

                this.flip.x ? this.x = -Math.abs(this.x) : this.x = Math.abs(this.x)
                this.flip.y ? this.y = -Math.abs(this.y) : this.y = Math.abs(this.y)
                this.ctxConfigDynamic()
                console.log(this.x)
                callback(now)

                if (this.scene.debugMode) {
                    this.scene.ctx?.fillRect(
                        this.x, this.y,
                        this.width, this.height
                    )
                }

                this.scene.ctx?.drawImage(
                    this.scene.img,
                    this.atlas.x, this.atlas.y,
                    this.atlas.w, this.atlas.h,
                    this.x, this.y,
                    this.width, this.height
                )
                prevXY = { x: this.x, y: this.y }
            })
        }
    }

    addAnim(key: string, atlasArray: atlas[]) {
        this.animArray.push({ key: key, imgSrc: this.imgSrc, atlasArray })
    }

    playAnim(key: string, fps = 0) {
        const anim = this.animArray.find(a => a.key === key)
        if (!anim) return
        this.imgSrc = anim.imgSrc

        const frame = fps === 0 ? anim.atlasArray.length : fps
        let i = 0
        let lastT = 0
        let interval = 1000 / frame

        if (this.prevAnimKey !== key) {
            this.redraw((now) => {
                if (i > anim.atlasArray.length - 1) i = 0
                this.atlas = {
                    x: anim.atlasArray[i].x, y: anim.atlasArray[i].y,
                    w: anim.atlasArray[i].w, h: anim.atlasArray[i].h
                }

                const deltaT = now - lastT
                if (deltaT > interval) {
                    i++
                    lastT = now - (deltaT % interval)
                }
            })
        }
        this.prevAnimKey = key
    }
}

export type atlas = { x: number, y: number, w: number, h: number }


animation(fps: number, imgAtlasArray: imgAtlas[]) {
        let i = 0
        let lastT = 0
        let interval = 1000 / fps

        const anim = (currentT: number) => {
            if (!this.scene.ctx) return

            if (i > imgAtlasArray.length - 1) i = 0
            this.scene.ctx.clearRect(0, 0, 100, 100) // TODO
            this.scene.ctx.drawImage(
                this.scene.img,
                imgAtlasArray[i].sX, imgAtlasArray[i].sY,
                imgAtlasArray[i].sW, imgAtlasArray[i].sH,
                imgAtlasArray[i].dX - (-this.x), imgAtlasArray[i].dY - (-this.y),
                imgAtlasArray[i].dW, imgAtlasArray[i].dH
            )

            const deltaT = currentT - lastT
            if (deltaT > interval) {
                i++
                lastT = currentT - (deltaT % interval)
            }

            requestAnimationFrame(anim)
        }

        this.scene.img.onload = () => {
            requestAnimationFrame(anim)
        }
    }
