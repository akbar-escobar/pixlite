export class Input {
    eventMap = new Map<eventKey, (e: eventMap[eventKey]) => void>()
    constructor() {
        window.addEventListener("keydown", (e) => {
            this.eventMap.get("keyboard")!(e)
        })
        window.addEventListener("pointerdown", (e) => {
            this.eventMap.get("pointer")!(e)
        })
    }

    event<K extends eventKey>(key: K, callback: (e: eventMap[K]) => void) {
        this.eventMap.set(key, callback)
    }
}

type eventMap = {
    pointer: PointerEvent,
    keyboard: KeyboardEvent
}
type eventKey = keyof eventMap
